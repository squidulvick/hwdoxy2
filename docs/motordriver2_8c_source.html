15void set_duty(motor_t*p_mot, int16_t duty)
16{
17 if (duty<0){
18 //if duty is negative, set fwd pin to high and drive PWM to negative pin
19 __HAL_TIM_SET_COMPARE(p_mot->htim,p_mot->fwd_channel,p_mot->high); //set forward channel high
20 __HAL_TIM_SET_COMPARE(p_mot->htim,p_mot->bwd_channel,duty); //set PWM in backward channel
21 printf("Setting Duty Cycle to %d\n",duty);
22 } else {
23 //else duty is positive and in range
24 __HAL_TIM_SET_COMPARE(p_mot->htim,p_mot->bwd_channel,p_mot->high); //set backward channel high
25 __HAL_TIM_SET_COMPARE(p_mot->htim,p_mot->fwd_channel,duty); //send PWM in forward channel
26 printf("Setting Duty Cycle to %d\n",duty);
27 }

29}
